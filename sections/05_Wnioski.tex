\section{Wnioski}

\begin{frame}{Wnioski}
    \begin{enumerate}
        \item<1-> \highlight{Wykorzystanie OZE}: Polska, pomimo posiadania istotnego udziału OZE w mocy zainstalowanej, nie jest w stanie wykorzystać pełni jej potencjału.
        \item<2-> \highlight{Rozwój OZE}: Trend wzrostowy łącznej mocy zainstalowanej OZE budzi konieczność prędkiego znalezienia sposobów na wykorzystanie tych zasobów.
        \item<3-> \highlight{Emisje}: Rozwój źródeł fotowoltaicznych pozwala na ograniczenie emisji.
        \item<4-> \highlight{Ceny za energię}: Przy właściwych warunkach, źródła energii odnawialnej pomagają znacząco zredukować koszty energii.
    \end{enumerate}
\end{frame}
