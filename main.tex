% ============================================================================
% Document Class and Basic Setup
% ============================================================================
% Defines the document type as a 16:9 widescreen beamer presentation
\documentclass[
    compress,          % Compresses the navigation bar for a cleaner look
    12pt,              % Sets the base font size to 12 points
    aspectratio=169,     % Sets the slide aspect ratio to 16:9 (widescreen)
    xcolor=table,      % Enables advanced color options for tables
    % To show frame numbers, uncomment the following line:
    % frameno
]{beamer}

% ============================================================================
% Configuration Files and Core Setup
% ============================================================================

% --- Core Setup ---
% Loads main document settings (e.g., fonts, hyphenation)
\input{config/document_setup}
% Loads all necessary LaTeX packages for the presentation
\input{config/packages}

% --- Theme and Appearance ---
% CORRECTED VERSION:
% This prepends the local 'theme/' folder to LaTeX's input search path.
% It uses \edef to prevent the infinite loop that caused the capacity error.
\makeatletter
\edef\input@path{{\string./theme/}\input@path}
\makeatother

% Now, LaTeX can find and load the 'Arguelles' theme from your local directory.
\usetheme{Arguelles}

% Loads project-specific color definitions (e.g., for charts and graphs)
\input{config/colors}


% --- Document Content Setup ---
% Loads presentation metadata (title, subtitle, author, date, etc.)
\input{config/metadata}
% Loads custom commands and environments for cleaner code in the main file
\input{config/custom_commands}

% ============================================================================
%                           START OF DOCUMENT
% ============================================================================
\begin{document}

% ============================================================================
% Frontmatter
% ============================================================================

% Display the title page using metadata from 'config/metadata.tex'
% The 'plain' option removes beamer's header and footer for a clean look
\begin{frame}[plain, noframenumbering]
    \titlepage
\end{frame}

% Display the table of contents
% 'currentsection' highlights the current section in the table of contents
\begin{frame}{Plan prezentacji}
    \tableofcontents[currentsection]
\end{frame}

% ============================================================================
% Main Content Sections
% ============================================================================

% Each section is included from its own .tex file using a custom command.
% This keeps the main file organized and easy to manage.
\includesection{01_Cele prezentacji}
\includesection{02_Dane statystyczne}
\includesection{03_Wykorzystane narzędzia i biblioteki}
\includesection{04_Przedstawienie danych}
\includesection{05_Wnioski}
\includesection{06_Literatura}
\includesection{07_Podziękowania}

% ============================================================================
% Appendix (Optional)
% ============================================================================
\appendix
% You can place additional, non-essential slides here.
% For example:
% \begin{frame}{Dodatkowe dane}
%     Content for the appendix...
% \end{frame}

% ============================================================================
%                            END OF DOCUMENT
% ============================================================================
\end{document}
